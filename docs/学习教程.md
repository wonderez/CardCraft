```python
from PyQt6.QtGui import QSyntaxHighlighter, QTextCharFormat
from PyQt6.QtCore import QRegularExpression

class MarkdownHighlighter(QSyntaxHighlighter):
    """Markdown语法高亮器"""
    
    def __init__(self, parent=None):
        super().__init__(parent)
        self.highlighting_rules = []
        self.setup_rules()
    
    def setup_rules(self):
        """设置高亮规则"""
        # 标题格式
        title_format = QTextCharFormat()
        title_format.setForeground(QColor("#0F79D8"))
        title_format.setFontWeight(QFont.Weight.Bold)
        
        # 添加标题规则
        self.highlighting_rules.append(
            (QRegularExpression(r"^#{1,6}\s.*"), title_format)
        )
        
        # 粗体格式
        bold_format = QTextCharFormat()
        bold_format.setForeground(QColor("#D73A49"))
        bold_format.setFontWeight(QFont.Weight.Bold)
        
        # 添加粗体规则
        self.highlighting_rules.append(
            (QRegularExpression(r"\*\*[^\*]+\*\*"), bold_format)
        )
    
    def highlightBlock(self, text):
        """应用高亮规则"""
        for pattern, format in self.highlighting_rules:
            iterator = pattern.globalMatch(text)
            while iterator.hasNext():
                match = iterator.next()
                self.setFormat(match.capturedStart(), 
                              match.capturedLength(), 
                              format)
```

### 4.2 分页功能实现

#### 智能分页算法
```python
class SmartPaginator:
    """智能分页器"""
    
    def __init__(self, page_size="medium", font_size=18):
        self.page_sizes = {
            "small": (720, 960),
            "medium": (1080, 1440),
            "large": (1440, 1920)
        }
        
        self.page_width, self.page_height = self.page_sizes[page_size]
        self.font_size = font_size
        self.line_height = font_size * 1.6  # 行高
        self.margin = 60  # 页边距
        
        # 计算可用高度
        self.available_height = self.page_height - (2 * self.margin)
    
    def paginate(self, html_content):
        """将HTML内容分页"""
        # 简化的分页逻辑
        lines = html_content.split('\n')
        pages = []
        current_page = []
        current_height = 0
        
        for line in lines:
            # 估算行高
            line_height = self.estimate_line_height(line)
            
            # 检查是否超出页面高度
            if current_height + line_height > self.available_height:
                # 添加当前页
                pages.append('\n'.join(current_page))
                # 开始新页
                current_page = [line]
                current_height = line_height
            else:
                current_page.append(line)
                current_height += line_height
        
        # 添加最后一页
        if current_page:
            pages.append('\n'.join(current_page))
        
        return pages
    
    def estimate_line_height(self, line):
        """估算行高"""
        # 简单估算：标题行高更大
        if line.startswith('<h1>'):
            return self.line_height * 2
        elif line.startswith('<h2>'):
            return self.line_height * 1.8
        elif line.startswith('<h3>'):
            return self.line_height * 1.6
        else:
            return self.line_height
```

from PyQt6.QtPrintSupport import QPrinter
from PyQt6.QtGui import QPainter

class ImageExporter:
    """图片导出器"""
    
    def __init__(self, web_view):
        self.web_view = web_view
    
    def export_to_image(self, output_path, width=1080, height=1440):
        """导出WebView内容为图片"""
        # 创建打印机
        printer = QPrinter(QPrinter.PrinterMode.HighResolution)
        printer.setOutputFormat(QPrinter.OutputFormat.PdfFormat)
        printer.setOutputFileName(output_path)
        printer.setPageSize(QPageSize(QPageSize.PageSizeId.A4))
        
        # 渲染到PDF
        self.web_view.render(printer)
        
        # 或者直接导出为图片
        image = QImage(width, height, QImage.Format.Format_ARGB32)
        image.fill(Qt.GlobalColor.white)
        
        painter = QPainter(image)
        self.web_view.render(painter)
        painter.end()
        
        image.save(output_path)

class PluginInterface(ABC):
    """插件接口"""
    
    @abstractmethod

    @abstractmethod

    @abstractmethod
    def initialize(self, application):
        """初始化插件"""
        pass

        """处理Markdown文本"""

#### 插件管理器
```python
import importlib
import os
from pathlib import Path

class PluginManager:
    """插件管理器"""
    
    def __init__(self):
        self.plugins = []
        self.plugin_dir = Path(__file__).parent / "plugins"
    
    def load_plugins(self):
        """加载所有插件"""
        if not self.plugin_dir.exists():
            return
        
        for file_path in self.plugin_dir.glob("*.py"):
            if file_path.name.startswith("__"):
                continue
            
            module_name = file_path.stem
            try:
                # 动态导入模块
                module = importlib.import_module(f"plugins.{module_name}")
                
                # 查找插件类
                for attr_name in dir(module):
                    attr = getattr(module, attr_name)
                    if (isinstance(attr, type) and 
                        issubclass(attr, PluginInterface) and 
                        attr != PluginInterface):
                        
                        # 创建插件实例
                        plugin = attr()
                        self.plugins.append(plugin)
                        print(f"加载插件: {plugin.get_name()} v{plugin.get_version()}")
            except Exception as e:
                print(f"加载插件 {module_name} 失败: {e}")
    
    def process_markdown(self, text):
        """使用所有插件处理Markdown"""
        result = text
        for plugin in self.plugins:
            result = plugin.process_markdown(result)
        return result
```

### 6.2 创建安装程序

#### 使用NSIS创建Windows安装程序
```nsis
; CardCraft.nsi
; 安装程序脚本

!define APPNAME "CardCraft"
!define VERSION "1.0.2.0"
!define PUBLISHER "CardCraft Studio"
!define URL "https://github.com/wonderez/CardCraft"

; 包含现代UI
!include "MUI2.nsh"

; 基本设置
Name "${APPNAME}"
OutFile "${APPNAME}_Setup_${VERSION}.exe"
InstallDir "$PROGRAMFILES\${APPNAME}"
InstallDirRegKey HKLM "Software\${APPNAME}" "InstallPath"
RequestExecutionLevel admin

; 界面设置
!define MUI_ABORTWARNING
!define MUI_ICON "resources\icons\app.ico"
!define MUI_UNICON "resources\icons\app.ico"

; 安装页面
!insertmacro MUI_PAGE_WELCOME
!insertmacro MUI_PAGE_LICENSE "LICENSE.txt"
!insertmacro MUI_PAGE_DIRECTORY
!insertmacro MUI_PAGE_INSTFILES
!insertmacro MUI_PAGE_FINISH

; 卸载页面
!insertmacro MUI_UNPAGE_WELCOME
!insertmacro MUI_UNPAGE_CONFIRM
!insertmacro MUI_UNPAGE_INSTFILES
!insertmacro MUI_UNPAGE_FINISH

; 语言
!insertmacro MUI_LANGUAGE "SimpChinese"

; 安装部分
Section "MainSection" SEC01
  SetOutPath "$INSTDIR"
  File /r "dist\CardCraft\*"
  
  ; 创建开始菜单快捷方式
  CreateDirectory "$SMPROGRAMS\${APPNAME}"
  CreateShortCut "$SMPROGRAMS\${APPNAME}\${APPNAME}.lnk" "$INSTDIR\CardCraft.exe"
  
  ; 创建桌面快捷方式
  CreateShortCut "$DESKTOP\${APPNAME}.lnk" "$INSTDIR\CardCraft.exe"
  
  ; 写入注册表
  WriteRegStr HKLM "Software\${APPNAME}" "InstallPath" "$INSTDIR"
  WriteRegStr HKLM "Software\${APPNAME}" "Version" "${VERSION}"
  
  ; 创建卸载程序
  WriteUninstaller "$INSTDIR\Uninstall.exe"
  WriteRegStr HKLM "Software\Microsoft\Windows\CurrentVersion\Uninstall\${APPNAME}" "DisplayName" "${APPNAME}"
  WriteRegStr HKLM "Software\Microsoft\Windows\CurrentVersion\Uninstall\${APPNAME}" "UninstallString" "$INSTDIR\Uninstall.exe"
  WriteRegStr HKLM "Software\Microsoft\Windows\CurrentVersion\Uninstall\${APPNAME}" "DisplayVersion" "${VERSION}"
  WriteRegStr HKLM "Software\Microsoft\Windows\CurrentVersion\Uninstall\${APPNAME}" "Publisher" "${PUBLISHER}"
  WriteRegStr HKLM "Software\Microsoft\Windows\CurrentVersion\Uninstall\${APPNAME}" "URLInfoAbout" "${URL}"
SectionEnd

; 卸载部分
Section "Uninstall"
  Delete "$INSTDIR\Uninstall.exe"
  RMDir /r "$INSTDIR"
  
  Delete "$SMPROGRAMS\${APPNAME}\${APPNAME}.lnk"
  RMDir "$SMPROGRAMS\${APPNAME}"
  Delete "$DESKTOP\${APPNAME}.lnk"
  
  DeleteRegKey HKLM "Software\Microsoft\Windows\CurrentVersion\Uninstall\${APPNAME}"
  DeleteRegKey HKLM "Software\${APPNAME}"
SectionEnd
```
